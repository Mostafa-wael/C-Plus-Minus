print "logic";
a = 1;
b = 0;
print a; // 1
print b; // 0
print a == b; // 0
print a != b; // 1
print a < b; // 0
print a > b; // 1
print a <= b; // 0
print a >= b; // 1
////////////////////////////////
print "More logic";
print !a; // 0
print a && b; // 0
print a || b; // 1
print (a && a && b) || a; // 1
////////////////////////////////
print "assignment";
a = 10; // 1010
b = 10; // 1010
c = 2; // 0010
d = a * b + c; 
print a == 10;
print b == 10;
print c == 2;
print d == 102; // 102
////////////////////////////////
print "arithmetic";
print -a == -10; // -10
// print -3-4;
print a+b == 20; // 20
print a-b == 0; // 0
print a/b == 1; // 1
print a%c == 0; // 0
print a * b - b + a / b == 91; // 91
////////////////////////////////
print "bitwise";
print a | c  == 10; // 1010 | 0010 = 1010 = 10
print a & c  == 2; // 1010 & 0010 = 0010 = 2
print a ^ c  == 8; // 1010 ^ 0010 = 1000 = 8
print a << c  == 40; // 1010 << 0010 = 101000 = 40
print a >> c == 2; // 1010 >> 0010 = 10 = 2
print ~a == -11; // -11
print (a<<c)>>c == 10; // 10
////////////////////////////////
